        /**
         * Update or create system value
         * @param $name
         * @param $value
         */
        public function setPropertyValue($name, $value)
        {
            if (is_array($value)) {
                $value = json_encode($value);
            }
            if (is_bool($value)) {
                if ($value) {
                    $value = 1;
                } else {
                    $value = 0;
                }
            }
            /** @var System $system */
            if (!($system = Yii::app()->system->getModel()->findByAttributes(["name" => $name]))) {
                $system = Yii::app()->system->getModel();
                $system->name = $name;
            }
            $system->value = $value;
            $system->savePropertyValue();
        }

        /**
         * Init ssl edit form
         *
         * @param SslConfigForm $form
         * @param bool $enduserPortal
         */
        public function initSslEditForm(SslConfigForm $form, $enduserPortal = false) : SslConfigForm
        {
            if (!$enduserPortal) {
                $form = $this->setSslEditForm($form);
                $form->notificationEmail = $this->lcrypt_notification_email;
                $form->mode = $this->lcrypt_mode;
            } else {
                $form->ssl = $this->enduser_portal_ssl;
                $form->domain = $this->enduser_portal_full_domain;
                $form->existingDomain = $this->enduser_portal_domain_ssl;
                if ($this->enduserPortalDomainSslObject && !$this->enduserPortalDomainSslObject->wildcard) {
                    $form->email = $this->enduserPortalDomainSslObject->email;
                    $form->error = $this->enduser_portal_ssl_generation_error;
                    $form->sslProvider = $this->enduserPortalDomainSslObject->provider;
                } else {
                    $form->sslProvider = SslManager::SSL_FORM_LIST;
                }
            }
            return $form;
        }
